export default function SDKBoxes({ items }) {
  return (
    <div className='row layout-boxes'>
      {items.map((box, index) => (
        <div className='column' key={index}>
          {/* Conditionally render the anchor tag based on box.available */}
          {box.available ? (
            <a href={box.link}>
              <div className='layout-boxes-item'>
                <div className={!('available' in box) || box.available ? 'layout-boxes-container' : 'layout-boxes-container-fade'}>
                  <h4>
                    {box.icon}
                    <span>{box.label}</span>
                  </h4>
                  {'available' in box && (
                    <div className='layout-boxes-available'>
                      <l className={box.available ? 'green' : 'grey'} r=''>
                        {box.available ? 'AVAILABLE' : 'COMING SOON'}
                      </l>
                    </div>
                  )}
                </div>
              </div>
            </a>
          ) : (
            <div className='layout-boxes-item'>
              <div className={!('available' in box) || box.available ? 'layout-boxes-container' : 'layout-boxes-container-fade'}>
                <h4>
                  {box.icon}
                  <span>{box.label}</span>
                </h4>
                {'available' in box && (
                  <div className='layout-boxes-available'>
                    <l className={box.available ? 'green' : 'grey'} r=''>
                      {box.available ? 'AVAILABLE' : 'COMING SOON'}
                    </l>
                  </div>
                )}
              </div>
            </div>
          )}
        </div>
      ))}
    </div>
  );
}
